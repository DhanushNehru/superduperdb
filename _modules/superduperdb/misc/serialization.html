[docs]class TorchSerializer(Serializer):

[docs]    @staticmethod
    def encode(object: t.Any, info: Info = None) -> bytes:
        if not isinstance(object, dict):
            was_gpu = str(device_of(object)) == 'cuda'
            object.to('cpu')
            f = io.BytesIO()
            torch.save(object, f)
            if was_gpu:
                object.to('cuda')
        else:
            f = io.BytesIO()
            torch.save(object, f)
        return f.getvalue()



[docs]    @staticmethod
    def decode(b: bytes, info: Info = None) -> t.Any:
        return torch.load(io.BytesIO(b))
