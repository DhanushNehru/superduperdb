from . import command
from superduperdb import ROOT
from superduperdb.misc import run
from superduperdb.misc.git import Git, git
import functools
import shutil
import sys

DOCS = 'docs'
DOCS_ROOT = ROOT / DOCS
APIDOC_ROOT = DOCS_ROOT / 'apidoc'
GH_PAGES = ROOT / '.cache/gh-pages'

run_gh = functools.partial(run.run, cwd=GH_PAGES)
out_gh = functools.partial(run.out, cwd=GH_PAGES)
git_gh = Git(out=out_gh)

def docs(commit_message: str):
    print('Building docs')

    APIDOC_ROOT.mkdir(exist_ok=True)

    if not GH_PAGES.exists():
        _make_gh_pages()
    elif git_gh.is_dirty():
        sys.exit(f'ERROR: docs: Your local changes in {GH_PAGES} would be overwritten')
    else:
        git_gh('pull')

    _clean()
    # run_gh(('sphinx-apidoc', '-f', '-o', str(APIDOC_ROOT), str(ROOT), 'superduperdb'))
    run_gh(('sphinx-build', '-a', str(DOCS_ROOT), '.'))

    git_gh('add', '.')
    git_gh('commit', '-m', commit_message)
    git_gh('push')



def _make_gh_pages():
    GH_PAGES.mkdir(parents=True)

    configs = git.configs()
    branches = git.branches()

    origin = configs['remote.origin.url']
    configs['remote.upstream.url']

    if 'gh-pages' in branches['origin']:
        git_gh('clone', '-b', 'gh-pages', origin, '.')
    else:
        git_gh('clone', origin, '.')
        git_gh('switch', 'gh-pages')
        git_gh('push', '--set-upstream', 'origin', 'testing')


def _clean():
    shutil.rmtree(APIDOC_ROOT)
    for i in GH_PAGES.iterdir():
        if i.name != '.git':
            if i.is_dir():
                shutil.rmtree(i)
            else:
                i.unlink()
