from superduperdb.misc.git import git

PUNCTUATION = '.,-_'


def _errors(msg):
    if len(msg) > 50:
        yield 'length greater than 50'
    if msg[0].islower():
        yield 'does not start with a capital letter'
    if ps := [p for p in PUNCTUATION if msg.endswith(p)]:
        yield f'ends with punctuation: "{ps[0]}"'
